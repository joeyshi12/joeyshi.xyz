I first organize all my markdown files within the /content folder. I further group my files by different courses I am taking.

I use Pandoc to transcompile each markdown file as a separate HTML page whose relative path in the target folder is the same as the markdown file in the content folder.

MD_FILES=$(shell find ./content -name \*.md)
HTML_FILES=$(MD_FILES:.md=.html)
BUILD_HTML_FILES=$(HTML_FILES:%=docs/%)

all: $(BUILD_HTML_FILES)

docs/%.html: %.md
        @mkdir -p $$(dirname $@)
        pandoc -s --mathjax -o $@ $<

new:
        $(if $(path),\
              @mkdir -p content/$$(dirname $(path));\
              cp -i templates/basic.md content/$(path),\
              $(error syntax: make new path=<content-path>)\
      )

clean:
        -rm -rf docs/content
